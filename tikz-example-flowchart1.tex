\usetikzlibrary{
  arrows.meta,
  chains,
  matrix,
  positioning,
}

\makeatletter
\def\@currentColor{white}

\tikzset{
  set color/.code = { \gdef\@currentColor{#1} },
  %% node
  box/.style = {
    draw,
    rounded corners=2mm,
    minimum width=4.5em,
    minimum height=2em,
    font=\color{white}\ttfamily,
    text height=1.5ex,
    text depth=.25ex,
    label={[black]above:#1},
  },
  box 1/.style = {box=#1, fill=orange},
  box 2/.style = {box=#1, fill=cyan},
  box 3/.style = {box=#1, fill=yellow!80!red},
  lower box/.style = {
    draw,
    fill=\@currentColor,
    minimum height=2.5em,
    inner xsep=2pt,
    font=\color{white}\ttfamily
  },
  %% matrix
  matrix default/.style = {
    matrix of nodes
  },
  upper matrix/.style = {
    matrix default,
    column sep=15pt
  },
  lower matrix/.style = {
    matrix default,
    column sep=0pt,
    set color=Green3, % set default color before key "lower box" is passed
    nodes=lower box
  },
  %% arrow
  arrow/.style = {
    arrows={ -Triangle[angle=45:2.5pt 3] },
    line width=1pt
  },
  %% chain
  every on chain/.style = { 
    join=by arrow
  },
}
\makeatother
